def create
  @ticket = @ticketable.tickets.new ticket_params
  @ticket.user = current_user

  if @ticket.save
    flash[:success] = "Seu Ticket foi gerado com sucesso."
  else
    flash[:error] = "Não foi possível gerar um Ticket."
  end
end
